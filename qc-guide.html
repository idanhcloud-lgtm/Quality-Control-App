🗄️ Hướng Dẫn Cài Đặt Database
Chạy SQL bên dưới trong Supabase → SQL Editor lần đầu tiên dùng
-- ① TẠO BẢNG create table if not exists qc_product_lines ( id uuid default gen_random_uuid() primary key, name text not null, sort_order int default 0, created_at timestamptz default now() ); create table if not exists qc_stages ( id uuid default gen_random_uuid() primary key, product_line_id uuid references qc_product_lines(id) on delete cascade, name text not null, sort_order int default 0, created_at timestamptz default now() ); create table if not exists qc_points ( id uuid default gen_random_uuid() primary key, stage_id uuid references qc_stages(id) on delete cascade, title text not null, description text default '', go_image_url text default '', nogo_image_url text default '', sort_order int default 0, created_at timestamptz default now() ); -- ② BẬT RLS + POLICY (cho phép đọc/ghi qua anon key) alter table qc_product_lines enable row level security; alter table qc_stages enable row level security; alter table qc_points enable row level security; create policy "read_all" on qc_product_lines for select using (true); create policy "write_all" on qc_product_lines for all using (true) with check (true); create policy "read_all" on qc_stages for select using (true); create policy "write_all" on qc_stages for all using (true) with check (true); create policy "read_all" on qc_points for select using (true); create policy "write_all" on qc_points for all using (true) with check (true); -- ③ DỮ LIỆU MẪU (chạy sau khi tạo bảng) insert into qc_product_lines (name, sort_order) values ('CARVER + STACKING', 1), ('TABLE', 2);
